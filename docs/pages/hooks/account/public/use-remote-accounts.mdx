# useRemoteAccounts [Hook for getting the proxy addresses of the remote accounts.]

## Import

```ts twoslash 
import { useRemoteAccounts } from '@abstract-money/react'
```

## Usage

```tsx twoslash [index.tsx]
import { useRemoteAccounts } from '@abstract-money/react'
import { stringToAccountId } from '@abstract-money/core'

function App() {
  const remoteAccounts = useRemoteAccounts({
    accountId: stringToAccountId('pion-1'),
    chainName: 'pion'
  })
}
```
