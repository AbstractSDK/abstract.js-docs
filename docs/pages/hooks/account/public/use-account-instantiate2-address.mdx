# useAccountInstantiate2Address [Hook for fetching the Instantiate2 address for an Abstract account]

## Import

```ts twoslash 
import { useAccountInstantiate2Address } from '@abstract-money/react'
```

## Usage

```tsx twoslash [index.tsx]
import { useAccountInstantiate2Address } from '@abstract-money/react'
import { stringToAccountId } from '@abstract-money/core'

function App() {
  const accountInstantiate2Address = useAccountInstantiate2Address({
    accountId: stringToAccountId('pion-1'),
    chainName: 'pion'
  })
}
```
