# useModules [Hook for getting modules installed on an account.]

## Import

```ts twoslash 
import { useModules } from '@abstract-money/react'
```

## Usage

```tsx twoslash [index.tsx]
import { useModules } from '@abstract-money/react'
import { stringToAccountId } from '@abstract-money/core'

function App() {
  const modulesQuery = useModules({
    accountId: stringToAccountId('pion-1'),
    chainName: 'pion',
  })
}
```
