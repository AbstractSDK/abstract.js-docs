# useRemoteAccountIds [Hook for getting account ids on remote chains.]

## Import

```ts twoslash 
import { useRemoteAccountIds } from '@abstract-money/react'
```

## Usage

```tsx twoslash [index.tsx]
import { useRemoteAccountIds } from '@abstract-money/react'
import { stringToAccountId } from '@abstract-money/core'

function App() {
  const remoteAccountIdsQuery = useRemoteAccountIds({
    accountId: stringToAccountId('pion-1'),
    chainName: 'pion',
  })
}
```
