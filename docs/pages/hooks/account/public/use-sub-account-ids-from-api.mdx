# useSubAccountIdsFromApi [Hook for getting sub accounts ids of an account.]

## Import

```ts twoslash 
import { useSubAccountIdsFromApi } from '@abstract-money/react'
```

## Usage

```tsx twoslash [index.tsx]
import { useSubAccountIdsFromApi } from '@abstract-money/react'
import { stringToAccountId } from '@abstract-money/core'

function App() {
  const subAccountIdsFromApi = useSubAccountIdsFromApi({
    accountId: stringToAccountId('pion-1'),
    chainName: 'pion',
  })
}
```
